# Pensadero NODO — Reconocimiento facial
#
# Dependencias Python para el módulo de detección y reconocimiento facial
# (face_detector.py). Instalar dentro de un venv local del proyecto:
#
#   cd backend/python
#   python -m venv .venv
#   .venv\Scripts\activate    # Windows
#   pip install -r requirements.txt
#
# Usamos `onnxruntime-gpu`: en máquinas con NVIDIA CUDA aprovecha la GPU
# (~10x más rápido, 1s/imagen vs 8s en CPU). En máquinas sin GPU compatible
# cae a CPU automáticamente; el código de face_detector.py detecta los
# providers disponibles y los elige en orden CUDA → CPU.
#
# Si la instalación de onnxruntime-gpu falla en algún equipo (incompatibilidad
# de CUDA, drivers viejos, etc.), sustituir por `onnxruntime` a secas y todo
# seguirá funcionando, solo más lento.

insightface==0.7.3
onnxruntime-gpu>=1.18
numpy>=1.24
opencv-python>=4.8
pillow>=10.0

# CLIP / M-CLIP — embeddings visuales + texto multilingue para
# place recognition, image search y (a futuro) text-to-image en español.
# torch+transformers son ~2 GB; multilingual-clip añade el text encoder
# XLM-RoBERTa-Large. Total disco ~3 GB, VRAM cargado ~2.5 GB.
torch>=2.0
transformers>=4.38
multilingual-clip>=1.0.10
sentencepiece>=0.2  # tokenizer XLM-RoBERTa
ftfy
regex
